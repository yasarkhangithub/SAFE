058/**
059 * An implementation of RepositoryConnection that uses {@link FederationEvalStrategy} to evaluate 
060 * provided queries. Prior to evaluation various optimizations are performed, see {@link Optimizer}
061 * for further details.
062 * 
063 * Implementation notes:
064 *  - the federation connection currently is read only
065 *  - not all methods are implemented as of now
066 * 
067 * @author Andreas Schwarte
068 *
069 */
070public class FedXConnection extends SailConnectionBase {
071        
072        public static Logger log = Logger.getLogger(FedXConnection.class);
073        protected FedX federation;
074        
075        public FedXConnection(FedX federation)
076                        throws SailException {
077                super(new SailBaseDefaultImpl());
078                this.federation = federation;
079        }

081        @Override
082        protected CloseableIteration<? extends BindingSet, QueryEvaluationException> evaluateInternal(
083                        TupleExpr query, Dataset dataset, BindingSet bindings,
084                        boolean includeInferred) throws SailException {
085                
086                FederationEvalStrategy strategy = FederationManager.getInstance().getStrategy();
087
088                long start=0;
089                if (true) {
090                        if (log.isDebugEnabled()) {
091                                log.debug("Optimization start");
092                                start = System.currentTimeMillis();
093                        }
094                        try {
095                                String queryString = getOriginalQueryString(bindings);
096                                if (queryString==null)
097                                        logger.warn("Query string is null. Please check your FedX setup.");
098                                QueryInfo queryInfo = new QueryInfo(queryString, getOriginalQueryType(bindings));
099                                FederationManager.getMonitoringService().monitorQuery(queryInfo);
100                                query = Optimizer.optimize(query, dataset, bindings, strategy, queryInfo);
101                        }  catch (Exception e) {
102                                log.error("Exception occured during optimization.", e);
103                                throw new SailException(e);
104                        }
105                        if (log.isDebugEnabled())
106                                log.debug(("Optimization duration: " + ((System.currentTimeMillis()-start))));
107                }
108
109                // log the optimized query plan, if Config#isLogQueryPlan(), otherwise void operation
110                FederationManager.getMonitoringService().logQueryPlan(query);
111                
112                if (Config.getConfig().isDebugQueryPlan()) {
113                        System.out.println("Optimized query execution plan: \n" + query);
114                        log.debug("Optimized query execution plan: \n" + query);
115                }
116                
117                try {
118                        return strategy.evaluate(query, EmptyBindingSet.getInstance());
119                } catch (QueryEvaluationException e) {
120                        throw new SailException(e);
121                }               
122        }

149        @Override
150        protected CloseableIteration<? extends Resource, SailException> getContextIDsInternal() throws SailException {
151                
152                final WorkerUnionBase<Resource> union = new SynchronousWorkerUnion<Resource>(new QueryInfo("getContextIDsInternal", QueryType.UNKNOWN));    
153                
154                for (final Endpoint e : federation.getMembers()) {
155                        union.addTask( new ParallelTask<Resource>() {
156                                @Override
157                                public CloseableIteration<Resource, QueryEvaluationException> performTask()       throws Exception {
158                                        return new RepositoryExceptionConvertingIteration<Resource>(e.getConn().getContextIDs());
159                                }                               
160                                @Override
161                                public ParallelExecutor<Resource> getControl() {
162                                        return union;
163                                }
164                        });
165                }
166                
167                // execute the union in a separate thread
168                FederationManager.getInstance().getExecutor().execute(union);
169                
170                return new DistinctIteration<Resource, SailException>( new ExceptionConvertingIteration<Resource, SailException>(union) {
171                        @Override
172                        protected SailException convert(Exception e) {
173                                return new SailException(e);
174                        }                       
175                });
176        }

189        @Override
190        protected CloseableIteration<? extends Statement, SailException> getStatementsInternal(
191                        Resource subj, URI pred, Value obj, final boolean includeInferred,
192                        final Resource... contexts) throws SailException {
193                        
194                try {
195                        FederationEvalStrategy strategy = FederationManager.getInstance().getStrategy();
196                        QueryInfo queryInfo = new QueryInfo(subj, pred, obj);
197                        FederationManager.getMonitoringService().monitorQuery(queryInfo);
198                        CloseableIteration<Statement, QueryEvaluationException> res = strategy.getStatements(queryInfo, subj, pred, obj, contexts);
199                        return new ExceptionConvertingIteration<Statement, SailException>(res) {
200                                @Override
201                                protected SailException convert(Exception e) {
202                                        return new SailException(e);
203                                }
204                        };                      
205                } catch (RuntimeException e) {
206                        throw e;
207                } catch (Exception e) {
208                        throw new SailException(e);
209                }
210        }

239        @Override
240        protected long sizeInternal(Resource... contexts) throws SailException {
241                if (contexts!=null && contexts.length>0)
242                        throw new UnsupportedOperationException("Context handling for size() not supported");
243                long size = 0;
244                List<String> errorEndpoints = new ArrayList<String>();
245                for (Endpoint e : federation.getMembers()) {
246                        try     {
247                                size += e.size();
248                        } catch (RepositoryException e1){
249                                errorEndpoints.add(e.getId());
250                        }
251                }
252                if (errorEndpoints.size()>0)
253                        throw new SailException("Could not determine size for members " + errorEndpoints.toString() + 
254                                        "(Supported for NativeStore and RemoteRepository only). Computed size: " +size);
255                return size;
256        }

263        private static String getOriginalQueryString(BindingSet b) {
264                if (b==null)
265                        return null;
266                Value q = b.getValue(FedXSailRepositoryConnection.BINDING_ORIGINAL_QUERY);
267                if (q!=null)
268                        return q.stringValue();
269                return null;
270        }
271        
272        private static QueryType getOriginalQueryType(BindingSet b) {
273                if (b==null)
274                        return null;
275                Value q = b.getValue(FedXSailRepositoryConnection.BINDING_ORIGINAL_QUERY_TYPE);
276                if (q!=null)
277                        return QueryType.valueOf(q.stringValue());
278                return null;
279        }
