022
023import org.apache.log4j.Logger;

027

034/**
035 * Structure to maintain endpoint information, e.g. type, location, etc. For initialization the
036 * underlying repository must be set using {@link #setRepo(Repository, TripleSource, boolean)}.<p>
037 * 
038 * All endpoints need to be added to the {@link EndpointManager}, moreover endpoints can be 
039 * looked up using their id and their connection.<p>
040 * 
041 * An endpoint uses a Singleton for the repository connection. If by chance this connection is broken,
042 * e.g. due to a SocketException, a call to {@link #repairConnection()} reinitializes the connection.<p>
043 * 
044 * Note: Interaction with endpoints should be done via the EndpointManager
045 * 
046 * @author Andreas Schwarte
047 * @see EndpointManager
048 */
049public class Endpoint  {
050        
051        public static Logger log = Logger.getLogger(Endpoint.class);
052        
053        /**
054         * Classify endpoints into remote or local ones.
055         * 
056         * @author Andreas Schwarte
057         */
058        public static enum EndpointClassification { Local, Remote; };
059        public static enum EndpointType {
060                NativeStore(Arrays.asList("NativeStore", "lsail/NativeStore")), 
061                SparqlEndpoint(Arrays.asList("SparqlEndpoint", "api/sparql")), 
062                RemoteRepository(Arrays.asList("RemoteRepository")), 
063                Other(Arrays.asList("Other"));
064                
065                private List<String> formatNames;
066                private EndpointType(List<String> formatNames) {
067                        this.formatNames = formatNames;
068                }       
069                
070                /**
071                 * Returns true if the endpoint type supports the
072                 * given format (e.g. mime-type). Consider as an
073                 * example the SparqlEndpoint which supports
074                 * format "api/sparql".
075                 * @param format
076                 * @return
077                 */
078                public boolean supportsFormat(String format) {
079                        return formatNames.contains(format);
080                }
081                
082                /**
083                 * returns true if the given format is supported by
084                 * some repository type.
085                 * 
086                 * @param format
087                 * @return
088                 */
089                public static boolean isSupportedFormat(String format) {
090                        if (format==null)
091                                return false;
092                        for (EndpointType e  : values())
093                                if (e.supportsFormat(format))
094                                        return true;
095                        return false;
096                }
097        }
098        
099        protected String id = null;                                                                             // the identifier
100        protected String name = null;                                                                   // the name
101        protected String endpoint = null;                                                               // the endpoint, e.g. for SPARQL the URL
102        protected EndpointType type = null;                                                                     // the type, e.g. SPARQL, NativeRepo
103        protected EndpointClassification endpointClassification;                // the endpoint classification
104                
105        protected Repository repo;
106        protected RepositoryConnection conn  = null;    // a Singleton RepositoryConnection for the given endpoint
107        protected boolean initialized = false;                  // true, iff the contained repository is initialized
108        protected TripleSource tripleSource;                    // the triple source, initialized when repository is set
109        protected EndpointConfiguration endpointConfiguration;  // additional endpoint type specific configuration
110
111        /**
112         * Construct a new endpoint.
113         * 
114         * @param id
115         *                      the globally unique identifier, e.g. SPARQL_dbpedia351
116         * @param name
117         *                      the name of this endpoint
118         * @param endpoint
119         *                      the endpoint, e.g. for SPARQL the URL
120         * @param type
121         *                      the type, e.g. SPARQL, NativeStore
122         */
123        public Endpoint(String id, String name, String endpoint, EndpointType type, EndpointClassification endpointClassification) {
124                super();
125                this.id = id;
126                this.name = name;
127                this.endpoint = endpoint;
128                this.type = type;
129                this.endpointClassification = endpointClassification;
130        }
131        
132        public String getName() {
133                return name;
134        }
135
136        public void setName(String name) {
137                this.name = name;
138        }
139
140        public Repository getRepo() {
141                return repo;
142        }
143
144        public TripleSource getTripleSource() {
145                return tripleSource;
146        }
147        
148        public EndpointClassification getEndpointClassification() {
149                return endpointClassification;
150        }
151
152        public boolean isLocal() {
153                return endpointClassification==EndpointClassification.Local;
154        }
155        
156        /**
157         * Set the underlying initialized repository.
158         * 
159         * @param repo
160         *                      the repository
161         * 
162         * @throws RepositoryException
163         */
164        public void setRepo(Repository repo) throws RepositoryException {
165                this.repo = repo;
166        }       
167        
168        
169        /**
170         * Additional endpoint specific configuration.
171         * 
172         * @return the endpointConfiguration
173         */
174        public EndpointConfiguration getEndpointConfiguration() {
175                return endpointConfiguration;
176        }
177
178        /**
179         * @param endpointConfiguration the endpointConfiguration to set
180         */
181        public void setEndpointConfiguration(EndpointConfiguration endpointConfiguration) {
182                this.endpointConfiguration = endpointConfiguration;
183        }
184
185        /**
186         * return a singleton connection object. this is valid for the whole lifetime of the
187         * underlying repository, i.e. until it is shutDown
188         * 
189         * @return
190         * 
191         * @throws RepositoryException
192         *                      if the repository is not initialized
193         */
194        public RepositoryConnection getConn() {
195                if (!initialized)
196                        throw new FedXRuntimeException("Repository for endpoint " + id + " not initialized");
197                return conn;
198        }
199
200        /**
201         *  @return
202         *      the identifier
203         */
204        public String getId() {
205                return id;
206        }
207
208        public void setId(String id) {
209                this.id = id;
210        }
211
212        /**
213         * Get the endpoint location, e.g. for SPARQL endpoints the url
214         * 
215         * @return
216         */
217        public String getEndpoint() {
218                return endpoint;
219        }
220
221        public void setEndpoint(String endpoint) {
222                this.endpoint = endpoint;
223        }
224
225        public EndpointType getType() {
226                return type;
227        }
228
229        public boolean isInitialized() {
230                return initialized;
231        }
232        
233        /**
234         * Returns the size of the given repository, i.e. the number of triples.
235         * Can only be applied for types NativeStore and RemoteRepository.
236         * 
237         * @return
238         * @throws RepositoryException
239         */
240        public long size() throws RepositoryException {
241                switch (type) {
242                case NativeStore:
243                case RemoteRepository:  
244                                return getConn().size();
245                default:                        
246                        throw new RepositoryException("Size() only supported for NativeStore and RemoteRepository.");
247                }
248        }
249        
250        /**
251         * Initialize this repository.
252         * 
253         * @throws RepositoryException
254         */
255        public void initialize() throws RepositoryException {
256                if (repo==null)
257                        throw new FedXRuntimeException("Repository for endpoint " + id + " not yet specified");
258                if (isInitialized())
259                        return;         
260                tripleSource = TripleSourceFactory.tripleSourceFor(this, type);
261                conn = repo.getConnection();
262                initialized = true;
263        }
264        
265        /**
266         * Repair the underlying connection, i.e. call repo.getConnection().
267         * 
268         * @return
269         *                      the new connection
270         * 
271         * @throws RepositoryException
272         *                              if a repository connection is thrown while creating the connection
273         */
274        public RepositoryConnection repairConnection() throws RepositoryException {
275                if (!initialized)
276                        throw new FedXRuntimeException("Repository for endpoint " + id + " not initialized");
277
278                log.debug("Repairing connection for endpoint " + id );
279                
280                if (conn!=null) {
281                        try {
282                                conn.close();
283                        } catch (RepositoryException e) { 
284                                log.warn("Connection of endpoint " + id + " could not be closed: " + e.getMessage());
285                        }
286                }
287                conn = repo.getConnection();
288                log.info("Connection for endpoint " + id + " successfully repaired.");
289                return conn;
290        }
291
292        /**
293         * Shutdown this repository.
294         * 
295         * @throws RepositoryException
296         */
297        public void shutDown() throws RepositoryException {
298                if (repo==null)
299                        throw new RepositoryException("Repository for endpoint " + id + " not yet specified");
300                if (!isInitialized())
301                        return;
302                conn.close();
303                conn = null;
304                repo.shutDown();
305                initialized = false;
306        }
307        
308        @Override
309        public int hashCode() {
310                final int prime = 31;
311                int result = 1;
312                result = prime * result + ((id == null) ? 0 : id.hashCode());
313                result = prime * result + ((type == null) ? 0 : type.hashCode());
314                return result;
315        }
316
317        @Override
318        public boolean equals(Object obj) {
319                if (this == obj)
320                        return true;
321                if (obj == null)
322                        return false;
323                if (getClass() != obj.getClass())
324                        return false;
325                Endpoint other = (Endpoint) obj;
326                if (id == null) {
327                        if (other.id != null)
328                                return false;
329                } else if (!id.equals(other.id))
330                        return false;
331                if (type == null) {
332                        if (other.type != null)
333                                return false;
334                } else if (!type.equals(other.type))
335                        return false;
336                return true;
337        }
338
339        @Override
340        public String toString() {
341                return "Endpoint [id=" + id + ", name=" + name + ", type=" + type + "]";
342        }
