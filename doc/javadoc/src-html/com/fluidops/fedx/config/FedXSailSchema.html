018package com.fluidops.fedx.config;
019
020import org.openrdf.model.URI;
021import org.openrdf.model.ValueFactory;
022import org.openrdf.model.impl.ValueFactoryImpl;
023
024/**
025 * Defines constants for the FedX schema which is used by {@link FedXSailFactory}
026 * to initialize FedX federations.
027 * 
028 * @author Andreas Schwarte
029 */
030public class FedXSailSchema {
031
032        
033        /** FedX schema namespace (<tt>http://www.fluidops.com/config/fedx#</tt>). */
034        public static final String NAMESPACE = "http://www.fluidops.com/config/fedx#";
035        
036        /** <tt>http://www.fluidops.com/config/fedx#fedxConfig</tt>  */
037        public final static URI fedXConfig;
038        
039        static {
040                ValueFactory factory = ValueFactoryImpl.getInstance();
041                fedXConfig = factory.createURI(NAMESPACE, "fedxConfig");
042        }
043}
