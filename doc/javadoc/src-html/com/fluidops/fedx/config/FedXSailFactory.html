018package com.fluidops.fedx.config;
019
020import java.util.Collections;
021
022import org.openrdf.sail.Sail;
023import org.openrdf.sail.config.SailConfigException;
024import org.openrdf.sail.config.SailFactory;
025import org.openrdf.sail.config.SailImplConfig;
026
027import com.fluidops.fedx.FedXFactory;
028import com.fluidops.fedx.FederationManager;
029import com.fluidops.fedx.exception.FedXException;
030import com.fluidops.fedx.structures.Endpoint;
031
032/**
033 * A {@link SailFactory} that initializes FedX Sails based on 
034 * the provided configuration data.
035 * 
036 * @author Andreas Schwarte
037 *
038 */
039public class FedXSailFactory implements SailFactory {
040
041        /**
042         * The type of repositories that are created by this factory.
043         * 
044         * @see SailFactory#getSailType()
045         */
046        public static final String SAIL_TYPE = "fluidops:FedX";
047        
048        
049        @Override
050        public SailImplConfig getConfig() {
051                return new FedXSailConfig();
052        }
053
054        @Override
055        public Sail getSail(SailImplConfig config) throws SailConfigException   {
056                
057                if (!SAIL_TYPE.equals(config.getType())) {
058                        throw new SailConfigException("Invalid Sail type: " + config.getType());
059                }
060                
061                if (!(config instanceof FedXSailConfig)) {
062                        throw new SailConfigException("FedXSail config expected, was " + config.getClass().getCanonicalName());
063                }       
064                
065                FedXSailConfig fedXSailConfig = (FedXSailConfig)config;
066                String fedxConfig = fedXSailConfig.getFedxConfig();
067                
068                if (fedxConfig==null)
069                        throw new SailConfigException("FedX Sail Configuration must not be null");
070                
071                try     {
072                        FedXFactory.initializeFederation(fedxConfig, Collections.<Endpoint>emptyList());
073                } catch (FedXException e) {
074                        throw new SailConfigException(e);
075                }
076                
077                return FederationManager.getInstance().getFederation();
078        }
079
080        /**
081         * Returns the Sail's type: <tt>fluidops:FedX</tt>.
082         */
083        @Override
084        public String getSailType()     {
085                return SAIL_TYPE;
086        }
087
088}
