061/**
062 * The FederationManager manages all modules necessary for the runtime behavior. This includes for 
063 * instance the federation layer instance, cache and statistics. It is Singleton and there can only 
064 * be on federation instance at a time.<p>
065 * 
066 * The factory {@link FedXFactory} provides various functions for initialization of FedX and should 
067 * be used as the entry point for any application using FedX.<p>
068 * 
069 * <code>
070 * Config.initialize(fedxConfig);
071 * List&ltEndpoint&gt members = ...                     // e.g. use EndpointFactory methods
072 * FedX fed = FedXFactory.initializeFederation(members);
073 * SailRepository repo = new SailRepository(fed);
074 * ReositoryConnection conn = repo.getConnection();
075 * 
076 * // Do something with the connection, e.g. query evaluation
077 * </code>
078 * 
079 * @author Andreas Schwarte
080 *
081 */
082public class FederationManager {
083

099        /**
100         * Initialize the Singleton {@link FederationManager} instance with the provided information. The
101         * {@link FederationManager} remains initialized until {@link #shutDown()} is invoked. 
102         * 
103         * @param members
104         *                              initialize the federation with a list of repository members, null and empty lists are allowed
105         * @param cache
106         *                              the cache instance to be used
107         * @param statistics
108         *                              the statistics instance to be used
109         */
110        public static SailRepository initialize(List<Endpoint> members, Cache cache, Statistics statistics) {
111                if (instance!=null)
112                        throw new FedXRuntimeException("FederationManager already initialized.");
113                
114                log.info("Initializing federation manager ...");
115                log.info("FedX Version Information: " + Version.getVersionString());            
116
117                monitoring = MonitoringFactory.createMonitoring();              
118                
119                Executor ex = Executors.newCachedThreadPool();
120                FedX federation = new FedX(members);
121
122                FedXSailRepository repo = new FedXSailRepository(federation);
123                
124                instance = new FederationManager(federation, cache, statistics, ex, repo);
125                instance.updateStrategy();
126                instance.reset();               
127                
128                try     {
129                        repo.initialize();
130                } catch (RepositoryException e) {
131                        // should never occur
132                        throw new FedXRuntimeException(e);      
133                }
134                
135                EndpointManager.initialize(members);
136                
137                if (Config.getConfig().isEnableJMX()) {
138                        try {
139                                MonitoringUtil.initializeJMXMonitoring();
140                        } catch (Exception e1) {
141                                log.error("JMX monitoring could not be initialized: " + e1.getMessage());
142                        }
143                }
144                
145                // initialize prefix declarations, if any
146                String prefixFile = Config.getConfig().getPrefixDeclarations();
147                if (prefixFile!=null) {
148                        QueryManager qm = instance.getQueryManager();
149                        Properties props = new Properties();
150                        try     {
151                                props.load(new FileInputStream(new File(prefixFile)));
152                        } catch (IOException e) {
153                                throw new FedXRuntimeException("Error loading prefix properties: " + e.getMessage());
154                        }
155                        
156                        for (String ns : props.stringPropertyNames()) {
157                                qm.addPrefixDeclaration(ns, props.getProperty(ns));     // register namespace/prefix pair
158                        }
159                }
160                
161                return repo;
162        }

225        /**
226         * Reset the {@link Scheduler} instances, i.e. abort all running threads and create a new
227         * scheduler instance.
228         */
229        public void reset() {
230                if (log.isDebugEnabled()) {
231                        log.debug("Scheduler for join and union operations initialized with " + Config.getConfig().getJoinWorkerThreads() + " and " + Config.getConfig().getUnionWorkerThreads() + " worker threads, respectively.");
232                }
233                
234                if (joinScheduler!=null)
235                        joinScheduler.abort();
236                joinScheduler = new ControlledWorkerScheduler<BindingSet>(Config.getConfig().getJoinWorkerThreads(), "Join Scheduler");           
237                
238                if (unionScheduler!=null)
239                        unionScheduler.abort();
240                unionScheduler = new ControlledWorkerScheduler<BindingSet>(Config.getConfig().getUnionWorkerThreads(), "Union Scheduler");                
241                
242        }

287        /**
288         * Add the specified endpoint to the federation. The endpoint must be initialized and
289         * the federation must not contain a member with the same endpoint location.
290         * 
291         * @param e
292         *                      the initialized endpoint
293         * @param updateStrategy
294         *                      optional parameter, to determine if strategy is to be updated, default=true
295         * 
296         * @throws FedXRuntimeException
297         *                       if the endpoint is not initialized, or if the federation has already a member with the
298         *           same location      
299         */
300        public void addEndpoint(Endpoint e, boolean ...updateStrategy) throws FedXRuntimeException {
301                log.info("Adding endpoint " + e.getId() + " to federation ...");
302                
303                /* check if endpoint is initialized*/
304                if (!e.isInitialized()) {
305                        try     {
306                                e.initialize();
307                        } catch (RepositoryException e1){
308                                throw new FedXRuntimeException("Provided endpoint was not initialized and could not be initialized: " + e1.getMessage(), e1);
309                        }
310                }
311                
312                /* check for duplicate before adding: heuristic => same location */
313                for (Endpoint member : federation.getMembers())
314                        if (member.getEndpoint().equals(e.getEndpoint()))
315                                throw new FedXRuntimeException("Adding failed: there exists already an endpoint with location " + e.getEndpoint() + " (eid=" + member.getId() + ")");
316        
317                federation.addMember(e);
318                EndpointManager.getEndpointManager().addEndpoint(e);
319                
320                if (updateStrategy==null || updateStrategy.length==0 || (updateStrategy.length==1 && updateStrategy[0]==true))
321                        updateStrategy();
322        }
323        
324        /**
325         * Add the specified endpoints to the federation and take care for updating all structures.
326         * 
327         * @param endpoints
328         *                              a list of initialized endpoints to add
329         */
330        public void addAll(List<Endpoint> endpoints) {
331                log.info("Adding " + endpoints.size() + " endpoints to the federation.");
332                
333                for (Endpoint e : endpoints) {
334                        addEndpoint(e, false);
335                }
336                
337                updateStrategy();
338        }
339        
340        /**
341         * Remove the specified endpoint from the federation.
342         * 
343         * @param e
344         *                      the endpoint
345         * @param updateStrategy
346         *                      optional parameter, to determine if strategy is to be updated, default=true
347         */
348        public void removeEndpoint(Endpoint e, boolean ...updateStrategy) throws RepositoryException {
349                log.info("Removing endpoint " + e.getId() + " from federation ...");
350                
351                /* check if e is a federation member */
352                if (!federation.getMembers().contains(e))
353                        throw new FedXRuntimeException("Endpoint " + e.getId() + " is not a member of the current federation.");
354                
355                federation.removeMember(e);
356                EndpointManager.getEndpointManager().removeEndpoint(e);
357                e.shutDown();
358                
359                if (updateStrategy==null || updateStrategy.length==0 || (updateStrategy.length==1 && updateStrategy[0]==true))
360                        updateStrategy();
361        }
362        
363        /**
364         * Remove all endpoints from the federation, e.g. to load a new preset. Repositories
365         * of the endpoints are shutDown, and the EndpointManager is added accordingly.
366         * 
367         * @throws RepositoryException
368         */
369        public void removeAll() throws RepositoryException {
370                log.info("Removing all endpoints from federation.");
371                
372                for (Endpoint e : new ArrayList<Endpoint>(federation.getMembers())) {
373                        removeEndpoint(e, false);
374                }
375                
376                updateStrategy();
377        }

379        /**
380         * return the number of triples in the federation as string. Retrieving
381         * the size is only supported {@link EndpointType#NativeStore} and
382         * {@link EndpointType#RemoteRepository}.
383         * 
384         * If the federation contains other types of endpoints, the size is
385         * indicated as a lower bound, i.e. the string starts with a 
386         * larger sign.
387         * @return
388         */
389        public String getFederationSize() {
390                long size = 0;
391                boolean isLowerBound = false;
392                for (Endpoint e : getFederation().getMembers())
393                        try     {
394                                size += e.size();
395                        } catch (RepositoryException e1) {
396                                isLowerBound = true;
397                        }
398                return isLowerBound ? ">" + size : Long.toString(size);
399        }

401        /**
402         * Shutdown the federation including the following operations: <p>
403         * 
404         * <ul>
405         *  <li>shut down repositories of all federation members</li>
406         *  <li>persist the cached information</li>
407         *  <li>clear the endpoint manager</li>
408         * </ul>
409         * 
410         * @throws FedXException
411         *                              if an error occurs while shutting down the federation
412         */
413        public void shutDown() throws FedXException {
414                log.info("Shutting down federation and all underlying repositories ...");
415                // Abort all running queries
416                federation.shutDownInternal();
417                cache.persist();
418                Config.reset();
419                EndpointManager.getEndpointManager().shutDown();
420                instance = null;
421                monitoring = null;
422        }

424        /**
425         * Create an appropriate worker union for this federation, i.e. a synchronous
426         * worker union for local federations and a multithreaded worker union
427         * for remote & hybrid federations.
428         * 
429         * @return
430         * 
431         * @see ControlledWorkerUnion
432         * @see SynchronousWorkerUnion
433         */
434        public WorkerUnionBase<BindingSet> createWorkerUnion(QueryInfo queryInfo) {
435                if (type==FederationType.LOCAL)
436                        return new SynchronousWorkerUnion<BindingSet>(queryInfo);
437                return new ControlledWorkerUnion<BindingSet>(unionScheduler, queryInfo);
438                
439        }

442        /**
443         * Update the federation evaluation strategy using the classification of endpoints 
444         * as provided by {@link Endpoint#getEndpointClassification()}:<p>
445         * 
446         * Which strategy is applied depends on {@link EvaluationStrategyFactory}.
447         * 
448         * Default strategies:
449         * <ul>
450         *  <li>local federation: {@link SailFederationEvalStrategy}</li>
451         *  <li>endpoint federation: {@link SparqlFederationEvalStrategy}</li>
452         *  <li>hybrid federation: {@link SparqlFederationEvalStrategy}</li>
453         * </ul>
454         * 
455         */
456        protected void updateStrategy() {
457                
458                int localCount=0, remoteCount=0;
459                for (Endpoint e : federation.getMembers()) {                    
460                        if (e.getEndpointClassification()==EndpointClassification.Remote) 
461                                remoteCount++;
462                        else 
463                                localCount++;           
464                }       
465                
466                boolean updated=false;
467                if (remoteCount==0) {
468                        if (type!=FederationType.LOCAL) {
469                                type = FederationType.LOCAL;
470                                updated=true;
471                        }
472                } else if (localCount==0) {
473                        if (type!=FederationType.REMOTE) {
474                                type = FederationType.REMOTE;
475                                updated=true;
476                        }
477                } else {
478                        if (type!=FederationType.HYBRID) {
479                                type = FederationType.HYBRID;
480                                updated=true;
481                        }                       
482                }
483                
484                if (updated) {
485                        strategy = EvaluationStrategyFactory.getEvaluationStrategy(type);               
486                        log.info("Federation updated. Type: " + type + ", evaluation strategy is " + instance.strategy.getClass().getSimpleName());
487                }
488                
489        }
