061        /**
062         * returns true iff there is at least one free variable, i.e. there is no binding
063         * for any variable
064         * 
065         * @param stmt
066         * @param bindings
067         * @return
068         */
069        public static boolean hasFreeVars(StatementPattern stmt, BindingSet bindings) {
070                for (Var var : stmt.getVarList()) {
071                        if(!var.hasValue() && !bindings.hasBinding(var.getName()))
072                                return true;    // there is at least one free var                               
073                }
074                return false;
075        }

077        /**
078         * Return the {@link Value} of the variable which is either taken from
079         * the variable itself (bound) or from the bindingsset (unbound).
080         * 
081         * @param var
082         * @param bindings
083         *                      the bindings, must not be null, use {@link EmptyBindingSet} instead
084         * 
085         * @return
086         *              the value or null
087         */
088        public static Value getVarValue(Var var, BindingSet bindings) {
089                if (var == null) {
090                        return null;
091                } else if (var.hasValue()) {
092                        return var.getValue();
093                } else {
094                        return bindings.getValue(var.getName());
095                }
096        }

103        public static StatementPattern toStatementPattern(Resource subj, URI pred, Value obj) {
104                Var s = subj==null ? new Var("s") : new Var("const_s", subj);
105                Var p = pred==null ? new Var("p") : new Var("const_p", pred);
106                Var o = obj==null ? new Var("o") : new Var("const_o", obj);
107                // TODO context
108                
109                return new StatementPattern(s, p, o);
110        }

127        /**
128         * Construct a SELECT query for the provided statement. 
129         * 
130         * @param stmt
131         * @param bindings
132         * @param filterExpr
133         * @param evaluated
134         *                      parameter can be used outside this method to check whether FILTER has been evaluated, false in beginning
135         * 
136         * @return
137         * @throws IllegalQueryException
138         */
139        public static TupleExpr selectQuery( StatementPattern stmt, BindingSet bindings, FilterValueExpr filterExpr, Boolean evaluated) throws IllegalQueryException {
140                
141                Set<String> varNames = new HashSet<String>();
142                TupleExpr expr = constructStatement(stmt, varNames, bindings);
143                                
144                if (varNames.size()==0)
145                        throw new IllegalQueryException("SELECT query needs at least one projection!");
146                                
147                if (filterExpr!=null) {
148                        try {
149                                expr = new Filter(expr, FilterUtils.toFilter(filterExpr));
150                                evaluated = true;
151                        } catch (FilterConversionException e) {
152                                log.warn("Filter could not be evaluated remotely. " + e.getMessage());
153                        }
154                }
155        
156                ProjectionElemList projList = new ProjectionElemList();
157                for (String var : varNames)
158                        projList.addElement( new ProjectionElem(var));
159                
160                Projection proj = new Projection(expr, projList);
161                
162                return proj;    
163        }

165        /**
166         * Construct a SELECT query for the provided {@link ExclusiveGroup}. Note that bindings
167         * and filterExpr are applied whenever possible.
168         *  
169         * @param group
170         *                              the expression for the query
171         * @param bindings
172         *                              the bindings to be applied
173         * @param filterExpr
174         *                              a filter expression or null
175         * @param evaluated
176         *                              parameter can be used outside this method to check whether FILTER has been evaluated, false in beginning
177         * 
178         * @return
179         */
180        public static TupleExpr selectQuery( ExclusiveGroup group, BindingSet bindings, FilterValueExpr filterExpr, Boolean evaluated) {
181                
182                
183                Set<String> varNames = new HashSet<String>();
184                List<ExclusiveStatement> stmts = group.getStatements();
185                
186                Join join = null;
187                
188                if (stmts.size()==2) {
189                        join = new Join(constructStatement(stmts.get(0), varNames, bindings), constructStatement(stmts.get(1), varNames, bindings));
190                } else {
191                        join = new Join();
192                        join.setLeftArg(constructStatement(stmts.get(0), varNames, bindings) );
193                        Join tmp = join;
194                        int idx;
195                        for (idx=1; idx<stmts.size()-1; idx++) {
196                                Join _u = new Join();
197                                _u.setLeftArg( constructStatement(stmts.get(idx), varNames, bindings) );
198                                tmp.setRightArg(_u);
199                                tmp = _u;
200                        }
201                        tmp.setRightArg( constructStatement(stmts.get(idx), varNames, bindings) );
202                }       
203                
204                TupleExpr expr = join;
205                                
206                if (filterExpr!=null) {
207                        try {
208                                expr = new Filter(expr, FilterUtils.toFilter(filterExpr));
209                                evaluated = true;
210                        } catch (FilterConversionException e) {
211                                log.warn("Filter could not be evaluated remotely. " + e.getMessage());
212                        }
213                }
214                
215                ProjectionElemList projList = new ProjectionElemList();
216                for (String var : varNames)
217                        projList.addElement( new ProjectionElem(var));
218                
219                Projection proj = new Projection(expr, projList);
220                
221                return proj;
222        }

224        /**
225         * Construct a SELECT query expression for a bound union.
226         * 
227         * Pattern:
228         * 
229         * SELECT ?v_1 ?v_2 ?v_N WHERE { { ?v_1 p o } UNION { ?v_2 p o } UNION ... } 
230         * 
231         * Note that the filterExpr is not evaluated at the moment.
232         * 
233         * @param stmt
234         * @param unionBindings
235         * @param filterExpr
236         * @param evaluated
237         *                      parameter can be used outside this method to check whether FILTER has been evaluated, false in beginning
238         * 
239         * @return
240         */
241        public static TupleExpr selectQueryBoundUnion( StatementPattern stmt, List<BindingSet> unionBindings, FilterValueExpr filterExpr, Boolean evaluated) {
242                
243                // TODO add FILTER expressions
244                
245                Set<String> varNames = new HashSet<String>();
246                
247                Union union = new Union();
248                union.setLeftArg(constructStatementId(stmt, Integer.toString(0), varNames, unionBindings.get(0)) );
249                Union tmp = union;
250                int idx;
251                for (idx=1; idx<unionBindings.size()-1; idx++) {
252                        Union _u = new Union();
253                        _u.setLeftArg( constructStatementId(stmt, Integer.toString(idx), varNames, unionBindings.get(idx)) );
254                        tmp.setRightArg(_u);
255                        tmp = _u;
256                }
257                tmp.setRightArg( constructStatementId(stmt, Integer.toString(idx), varNames, unionBindings.get(idx) ));
258                                
259                ProjectionElemList projList = new ProjectionElemList();
260                for (String var : varNames)
261                        projList.addElement( new ProjectionElem(var));
262                
263                Projection proj = new Projection(union, projList);
264
265                return proj;
266        }

270         * Construct a SELECT query for a grouped bound check.
271         * 
272         * Pattern:
273         * 
274         * SELECT DISTINCT ?o_1 .. ?o_N WHERE { { s1 p1 ?o_1 FILTER ?o_1=o1 } UNION ... UNION { sN pN ?o_N FILTER ?o_N=oN }}
275         * 
276         * @param stmt
277         * @param unionBindings
278         * @return
279         */

281                
282                Set<String> varNames = new HashSet<String>();
283                
284                Union union = new Union();
285                union.setLeftArg(constructStatementCheckId(stmt, 0, varNames, unionBindings.get(0)) );
286                Union tmp = union;
287                int idx;
288                for (idx=1; idx<unionBindings.size()-1; idx++) {
289                        Union _u = new Union();
290                        _u.setLeftArg( constructStatementCheckId(stmt, idx, varNames, unionBindings.get(idx)) );
291                        tmp.setRightArg(_u);
292                        tmp = _u;
293                }
294                tmp.setRightArg( constructStatementCheckId(stmt, idx, varNames, unionBindings.get(idx) ));
295                
296                ProjectionElemList projList = new ProjectionElemList();
297                for (String var : varNames)
298                        projList.addElement( new ProjectionElem(var));
299                
300                Projection proj = new Projection(union, projList);
301
302                return proj;
303        }

306        public static TupleExpr selectQueryIndependentJoinGroup(IndependentJoinGroup joinGroup, BindingSet bindings) {
307                
308                Set<String> varNames = new HashSet<String>();
309                
310                
311                Union union = null;
312                
313                if (joinGroup.getMemberCount()==2) {
314                        union = new Union();
315                        union.setLeftArg(constructStatementId((StatementPattern)joinGroup.getMembers().get(0), Integer.toString(0), varNames, bindings));
316                        union.setRightArg(constructStatementId((StatementPattern)joinGroup.getMembers().get(1), Integer.toString(1), varNames, bindings));
317                } else {
318                        union = new Union();
319                        union.setLeftArg(constructStatementId((StatementPattern)joinGroup.getMembers().get(0), Integer.toString(0), varNames, bindings) );
320                        Union tmp = union;
321                        int idx;
322                        for (idx=1; idx<joinGroup.getMemberCount()-1; idx++) {
323                                Union _u = new Union();
324                                _u.setLeftArg( constructStatementId((StatementPattern)joinGroup.getMembers().get(idx), Integer.toString(idx), varNames, bindings) );
325                                tmp.setRightArg(_u);
326                                tmp = _u;
327                        }
328                        tmp.setRightArg( constructStatementId((StatementPattern)joinGroup.getMembers().get(idx), Integer.toString(idx), varNames, bindings));
329                }               
330                
331                ProjectionElemList projList = new ProjectionElemList();
332                for (String var : varNames)
333                        projList.addElement( new ProjectionElem(var));
334                
335                Projection proj = new Projection(union, projList);
336
337                return proj;
338        }

341        /**
342         * Construct a select query representing a bound independent join group.
343         * 
344         * ?v_%stmt%#%bindingId$
345         * 
346         * SELECT ?v_0_0 ?v_0_1 ?v_1_0 ... 
347         *      WHERE { 
348         *              { ?v_0#0 p o UNION ?v_0_1 p o UNION ... } 
349         *              UNION 
350         *              { ?v_1_0 p o UNION ?v_1_1 p o UNION ... } 
351         *      UNION
352         *      ...
353         *  }
354         * @param joinGroup
355         * @param bindings
356         * @return
357         */
358        public static TupleExpr selectQueryIndependentJoinGroup(IndependentJoinGroup joinGroup, List<BindingSet> bindings) {
359                
360                Set<String> varNames = new HashSet<String>();
361                
362                
363                Union outer = null;
364                
365                if (joinGroup.getMemberCount()==2) {
366                        outer = new Union();
367                        outer.setLeftArg(constructInnerUnion((StatementPattern)joinGroup.getMembers().get(0), 0, varNames, bindings));
368                        outer.setRightArg(constructInnerUnion((StatementPattern)joinGroup.getMembers().get(1), 1, varNames, bindings));
369                } else {                        
370                        throw new RuntimeException("TODOO");
371                }               
372                
373                ProjectionElemList projList = new ProjectionElemList();
374                for (String var : varNames)
375                        projList.addElement( new ProjectionElem(var));
376                
377                Projection proj = new Projection(outer, projList);
378
379                return proj;
380        }

383        protected static Union constructInnerUnion(StatementPattern stmt, int outerID, Set<String> varNames, List<BindingSet> bindings) {
384                
385                Union union = new Union();
386                union.setLeftArg(constructStatementId(stmt, outerID + "_0", varNames, bindings.get(0)) );
387                Union tmp = union;
388                int idx;
389                for (idx=1; idx<bindings.size()-1; idx++) {
390                        Union _u = new Union();
391                        _u.setLeftArg( constructStatementId(stmt, outerID + "_" + idx, varNames, bindings.get(idx)) );
392                        tmp.setRightArg(_u);
393                        tmp = _u;
394                }
395                tmp.setRightArg( constructStatementId(stmt, outerID + "_" + idx, varNames, bindings.get(idx)));
396                
397                return union;
398        }

401        /**
402         * Construct the statement string, i.e. "s p o . " with bindings inserted wherever possible. Note that
403         * the free variables are added to the varNames set for further evaluation.
404         * 
405         * @param stmt
406         * @param varNames
407         * @param bindings
408         * 
409         * @return
410         */
411        protected static StatementPattern constructStatement(StatementPattern stmt, Set<String> varNames, BindingSet bindings) {
412                
413                Var subj = appendVar(stmt.getSubjectVar(), varNames, bindings);
414                Var pred =  appendVar(stmt.getPredicateVar(), varNames, bindings);
415                Var obj = appendVar(stmt.getObjectVar(), varNames, bindings);
416                
417                return new StatementPattern(subj, pred, obj);
418        }

420        /**
421         * Construct the statement string, i.e. "s p o . " with bindings inserted wherever possible. Variables
422         * are renamed to "var_"+varId to identify query results in bound queries. Note that
423         * the free variables are also added to the varNames set for further evaluation.
424         * 
425         * @param stmt
426         * @param varNames
427         * @param bindings
428         * 
429         * @return
430         */
431        protected static StatementPattern constructStatementId(StatementPattern stmt, String varID, Set<String> varNames, BindingSet bindings) {
432                
433                
434                Var subj = appendVarId(stmt.getSubjectVar(), varID, varNames, bindings);
435                Var pred =  appendVarId(stmt.getPredicateVar(), varID, varNames, bindings);
436                Var obj = appendVarId(stmt.getObjectVar(), varID, varNames, bindings);
437                
438                return new StatementPattern(subj, pred, obj);
439        }

441        /**
442         * Construct the statement string, i.e. "s p ?o_varID FILTER ?o_N=o ". This kind of statement
443         * pattern is necessary to later on identify available results.
444         * 
445         * @param stmt
446         * @param varID
447         * @param varNames
448         * @param bindings
449         * @return
450         */
451        protected static TupleExpr constructStatementCheckId(StatementPattern stmt, int varID, Set<String> varNames, BindingSet bindings) {
452                
453                String _varID = Integer.toString(varID);
454                Var subj = appendVarId(stmt.getSubjectVar(), _varID, varNames, bindings);
455                Var pred = appendVarId(stmt.getPredicateVar(), _varID, varNames, bindings);
456                
457                Var obj = new Var("o_" + _varID);
458                varNames.add("o_" + _varID);
459                                
460                Value objValue;
461                if (stmt.getObjectVar().hasValue()) {
462                        objValue = stmt.getObjectVar().getValue();
463                } else if (bindings.hasBinding(stmt.getObjectVar().getName())){
464                        objValue = bindings.getBinding(stmt.getObjectVar().getName()).getValue();
465                } else {
466                        // just to make sure that we see an error, will be deleted soon
467                        throw new RuntimeException("Unexpected.");
468                }
469                
470                Compare cmp = new Compare(obj, new ValueConstant(objValue));
471                cmp.setOperator(CompareOp.EQ);
472                Filter filter = new Filter( new StatementPattern(subj, pred, obj), cmp);
473                                
474                return filter;
475        }

480        /**
481         * Clone the specified variable and attach bindings.
482         *  
483         * @param sb
484         * @param var
485         * @param varNames
486         * @param bindings
487         * 
488         * @return
489
490         */
491        protected static Var appendVar(Var var, Set<String> varNames, BindingSet bindings) {
492                Var res = var.clone();
493                if (!var.hasValue()) {
494                        if (bindings.hasBinding(var.getName()))
495                                res.setValue( bindings.getValue(var.getName()) );
496                        else 
497                                varNames.add(var.getName());                    
498                }
499                return res;
500        }
501        
502        /**
503         * Clone the specified variable and attach bindings, moreover change name of variable
504         * by appending "_varId" to it.
505         * 
506         * @param sb
507         * @param var
508         * @param varNames
509         * @param bindings
510         * 
511         * @return
512         */
513        protected static Var appendVarId(Var var, String varID, Set<String> varNames, BindingSet bindings) {
514                Var res = var.clone();
515                if (!var.hasValue()) {
516                        if (bindings.hasBinding(var.getName())) {
517                                res.setValue( bindings.getValue(var.getName()) );
518                        } else {
519                                String newName = var.getName() + "_" + varID;
520                                varNames.add(newName);
521                                res.setName(newName);
522                        }                       
523                }
524                return res;
525        }
