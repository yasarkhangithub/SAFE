018package com.fluidops.fedx.evaluation;
019
020import com.fluidops.fedx.Config;
021import com.fluidops.fedx.FederationManager.FederationType;
022
023/**
024 * Factory class for retrieving the {@link FederationEvalStrategy} to be used
025 * 
026 * @author Andreas Schwarte
027 */
028public class EvaluationStrategyFactory {
029
030        
031        /**
032         * Return an instance of {@link FederationEvalStrategy} which is used for evaluating 
033         * the query. The type depends on the {@link FederationType} as well as on the 
034         * actual implementations given by the configuration, in particular this is
035         * {@link Config#getSailEvaluationStrategy()} and {@link Config#getSPARQLEvaluationStrategy()}.
036         * 
037         * @param federationType
038         * @return
039         */
040        public static FederationEvalStrategy getEvaluationStrategy(FederationType federationType) {
041                
042                switch (federationType) {
043                case LOCAL:             return instantiate(Config.getConfig().getSailEvaluationStrategy());
044                case REMOTE:
045                case HYBRID:
046                default:                return instantiate(Config.getConfig().getSPARQLEvaluationStrategy());
047                }
048        }
049        
050        private static FederationEvalStrategy instantiate(String evalStrategyClass) {
051                try {
052                        return (FederationEvalStrategy)Class.forName(evalStrategyClass).newInstance();
053                } catch (InstantiationException e) {
054                        throw new IllegalStateException("Class " + evalStrategyClass + " could not be instantiated.", e);
055                } catch (ClassNotFoundException e) {
056                        throw new IllegalStateException("Class " + evalStrategyClass + " could not be found, check whether the name is correct.", e);
057                } catch (Exception e) {
058                        throw new IllegalStateException("Unexpected error while instantiating " + evalStrategyClass + ":", e);
059                }
060        }
061}
