018package com.fluidops.fedx.evaluation.concurrent;
019
020import info.aduna.iteration.CloseableIteration;
021import info.aduna.iteration.LookAheadIteration;
022
023import org.apache.log4j.Logger;
024import org.openrdf.query.BindingSet;
025import org.openrdf.query.QueryEvaluationException;
026
027import com.fluidops.fedx.FederationManager;
028import com.fluidops.fedx.algebra.FedXService;
029import com.fluidops.fedx.evaluation.FederationEvalStrategy;
030
031
032/**
033 * Parallel executor for {@link FedXService} nodes, which wrap SERVICE expressions.
034 * 
035 * Uses the union scheduler to execute the task
036 * 
037 * @author Andreas Schwarte
038 */
039public class ParallelServiceExecutor extends LookAheadIteration<BindingSet, QueryEvaluationException> implements ParallelExecutor<BindingSet> {
040        
041        public static Logger log = Logger.getLogger(ParallelServiceExecutor.class);
042        
043        protected final FedXService service;
044        protected final FederationEvalStrategy strategy;
045        protected final BindingSet bindings;
046        
047        protected CloseableIteration<BindingSet, QueryEvaluationException> rightIter = null;
048        protected boolean finished = false;
049        protected Exception error = null;
050        
051        /**
052         * @param service
053         * @param strategy
054         * @param bindings
055         */
056        public ParallelServiceExecutor(FedXService service,
057                        FederationEvalStrategy strategy, BindingSet bindings) {
058                super();
059                this.service = service;
060                this.strategy = strategy;
061                this.bindings = bindings;
062        }
063
064        
065        @Override
066        public void run() {
067
068                ControlledWorkerScheduler<BindingSet> scheduler = FederationManager.getInstance().getUnionScheduler();
069                scheduler.schedule(new ParallelServiceTask());                  
070        }
071
072        @Override
073        public void addResult(CloseableIteration<BindingSet, QueryEvaluationException> res) {
074
075                synchronized (this) {
076                        
077                        rightIter = res;
078                        this.notify();
079                }
080                
081        }
082
083        @Override
084        public void toss(Exception e)   {
085
086                synchronized (this) {
087                        error = e;
088                }
089                
090        }
091
092        @Override
093        public void done()      {
094                ;       // no-op                
095        }
096
097        @Override
098        public boolean isFinished()     {
099                synchronized (this) {
100                        return finished;
101                }
102        }
103
104        @Override
105        public int getQueryId() {
106                return service.getQueryInfo().getQueryID();
107        }
108
109        @Override
110        protected BindingSet getNextElement() throws QueryEvaluationException {
111                
112                // error resulting from TOSS
113                if (error!=null) {
114                        if (error instanceof QueryEvaluationException)
115                                throw (QueryEvaluationException)error;
116                        throw new QueryEvaluationException(error);
117                }
118                        
119                if (rightIter==null) {  
120                        // block if not evaluated
121                        synchronized (this) {
122                                if (rightIter==null) {
123                                        try     {
124                                                // wait until the service expression is evaluated
125                                                this.wait();
126                                        }       catch (InterruptedException e)  {
127                                                log.debug("Interrupted exception while evaluating service.");
128                                        }
129                                }
130                        }
131                }
132                
133                if (rightIter.hasNext())
134                        return rightIter.next();                
135                
136                return null;
137        }
138
139
140        
141        /**
142         * Task for evaluating service requests
143         * 
144         * @author Andreas Schwarte
145         */
146        private class ParallelServiceTask implements ParallelTask<BindingSet> {
147        
148
149                @Override
150                public CloseableIteration<BindingSet, QueryEvaluationException> performTask()
151                                throws Exception {
152                        return strategy.evaluate(service.getService(), bindings);
153                }
154
155                @Override
156                public ParallelExecutor<BindingSet> getControl() {
157                        return ParallelServiceExecutor.this;
158                }
159                
160        }
161}
