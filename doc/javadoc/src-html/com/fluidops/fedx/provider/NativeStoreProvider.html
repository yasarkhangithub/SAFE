018package com.fluidops.fedx.provider;
019
020import java.io.File;
021
022import org.openrdf.query.algebra.evaluation.federation.FederatedServiceManager;
023import org.openrdf.repository.RepositoryException;
024import org.openrdf.repository.sail.SailRepository;
025import org.openrdf.sail.nativerdf.NativeStore;
026import org.openrdf.sail.nativerdf.NativeStoreExt;
027
028import com.fluidops.fedx.evaluation.SAILFederatedService;
029import com.fluidops.fedx.exception.FedXException;
030import com.fluidops.fedx.exception.FedXRuntimeException;
031import com.fluidops.fedx.structures.Endpoint;
032import com.fluidops.fedx.structures.Endpoint.EndpointClassification;
033import com.fluidops.fedx.util.FileUtil;
034
035
036/**
037 * Provider for an Endpoint that uses a Sesame {@link NativeStore} as underlying
038 * repository. For optimization purposes the NativeStore is wrapped within a
039 * {@link NativeStoreExt} to allow for evaluation of prepared queries without
040 * prior optimization. Note that NativeStores are always classified as 'Local'.
041 * 
042 * @author Andreas Schwarte
043 */
044public class NativeStoreProvider implements EndpointProvider {
045
046        @Override
047        public Endpoint loadEndpoint(RepositoryInformation repoInfo) throws FedXException {
048                
049                File store = FileUtil.getFileLocation(repoInfo.getLocation());
050                
051                if (!store.exists()){
052                        throw new FedXRuntimeException("Store does not exist at '" + repoInfo.getLocation() + ": " + store.getAbsolutePath() + "'.");
053                }
054                
055                try {
056                        NativeStore ns = new NativeStoreExt(store);
057                        SailRepository repo = new SailRepository(ns);
058                        repo.initialize();
059                        
060                        ProviderUtil.checkConnectionIfConfigured(repo);
061                        
062                        Endpoint res = new Endpoint(repoInfo.getId(), repoInfo.getName(), repoInfo.getLocation(), repoInfo.getType(), EndpointClassification.Local);
063                        res.setEndpointConfiguration(repoInfo.getEndpointConfiguration());
064                        res.setRepo(repo);
065                        
066                        // register a federated service manager to deal with this endpoint
067                        SAILFederatedService federatedService = new SAILFederatedService(res);
068                        federatedService.initialize();
069                        FederatedServiceManager.getInstance().registerService(repoInfo.getName(), federatedService);
070                        
071                        return res;
072                } catch (RepositoryException e) {
073                        throw new FedXException("Repository " + repoInfo.getId() + " could not be initialized: " + e.getMessage(), e);
074                }
075        }
076
077
078}
