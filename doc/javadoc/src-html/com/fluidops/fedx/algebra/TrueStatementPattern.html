018package com.fluidops.fedx.algebra;
019
020import org.openrdf.query.algebra.StatementPattern;
021
022/**
023 * A TrueStatementPattern represents a stmt with no free variables which in addition is
024 * available at one of the provided sources.
025 * 
026 * @author Andreas Schwarte
027 *
028 */
029public class TrueStatementPattern extends StatementPattern implements BoundJoinTupleExpr {
030
031        public TrueStatementPattern(StatementPattern node) {
032                super(node.getSubjectVar(), node.getPredicateVar(), node.getObjectVar(), node.getContextVar());
033        }
034}
